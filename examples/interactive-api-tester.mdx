## Overview

The interactive API tester provides a browser-based interface to test all GalioPay API endpoints. It maps 1:1 with the documented API routes and lets you modify payloads in real-time.

**Live Demo:** Navigate to `/examples` in the source application.

## Features

- Test all payment link and payment endpoints
- Edit JSON payloads directly in the UI
- Real-time response display
- Error handling with descriptive messages
- Pre-filled example payloads

## Implementation

### Main Interface

**Source:** `src/app/examples/page.tsx`

```tsx
"use client";

import { FormEvent, useState } from "react";

type ApiResult = {
  title: string;
  payload: unknown;
};

export default function ApiExamplesPage() {
  const [paymentLinkId, setPaymentLinkId] = useState("");
  const [proofToken, setProofToken] = useState("");
  const [paymentId, setPaymentId] = useState("");

  const [createPayload, setCreatePayload] = useState(
    JSON.stringify(initialCreatePayload, null, 2),
  );
  const [updatePayload, setUpdatePayload] = useState(
    JSON.stringify(initialUpdatePayload, null, 2),
  );
  const [refundPayload, setRefundPayload] = useState(
    JSON.stringify(initialRefundPayload, null, 2),
  );

  const [loading, setLoading] = useState<string | null>(null);
  const [result, setResult] = useState<ApiResult | null>(null);
  const [error, setError] = useState<string | null>(null);

  async function runRequest(
    title: string,
    url: string,
    method: "GET" | "POST" | "PUT",
    body?: unknown,
  ) {
    setLoading(title);
    setError(null);

    try {
      const response = await fetch(url, {
        method,
        headers: { "Content-Type": "application/json" },
        body: body ? JSON.stringify(body) : undefined,
      });

      const data = (await response.json()) as unknown;

      if (!response.ok) {
        throw new Error(
          typeof data === "object" &&
            data !== null &&
            "error" in data &&
            typeof (data as { error?: unknown }).error === "string"
            ? (data as { error: string }).error
            : "Request failed",
        );
      }

      setResult({ title, payload: data });
    } catch (err) {
      setError(err instanceof Error ? err.message : "Unexpected error");
    } finally {
      setLoading(null);
    }
  }

  // Endpoint handlers...
}
```

## Endpoint Details

### 1. Create Payment Link

**Method:** `POST /api/examples/payment-links`

```tsx
async function onCreatePaymentLink(event: FormEvent) {
  event.preventDefault();
  const body = JSON.parse(createPayload) as unknown;
  await runRequest(
    "Create payment link",
    "/api/examples/payment-links",
    "POST",
    body,
  );
}
```

**UI Component:**

```tsx
<form
  onSubmit={onCreatePaymentLink}
  className="rounded-2xl border border-line bg-card p-5"
>
  <h2 className="text-lg font-semibold text-foreground">
    POST /payment-links
  </h2>
  <textarea
    value={createPayload}
```

```tsx
</form>
```

### 3. Get Payment Link

**Method:** `GET /api/examples/payment-links/{id}?proof=`

```tsx
async function onGetPaymentLink(event: FormEvent) {
  event.preventDefault();

  if (!paymentLinkId || !proofToken) {
    setError("Payment Link ID and proof token are required");
    return;
  }

  await runRequest(
    "Get payment link",
    `/api/examples/payment-links/${paymentLinkId}?proof=${encodeURIComponent(proofToken)}`,
    "GET",
  );
}
```

**UI Component:**

```tsx
<form
  onSubmit={onGetPaymentLink}

    placeholder="payment-link-id"
```

## Response Display

All API responses are displayed in a unified response section:

```tsx
<section className="mt-6 rounded-2xl border border-line bg-card p-5">
  <h2 className="text-lg font-semibold text-foreground">Response</h2>
  {error ? <p className="mt-3 text-sm text-rose-600">{error}</p> : null}
  {result ? (
    <div className="mt-3">
      <p className="text-sm text-muted">{result.title}</p>
      <pre className="mt-2 overflow-x-auto rounded-xl bg-slate-900 p-4 font-mono text-xs text-slate-100">
        {JSON.stringify(result.payload, null, 2)}
      </pre>
    </div>
  ) : (
    <p className="mt-3 text-sm text-muted">
      Run any example to see the response JSON.
    </p>
  )}
</section>
```

## Error Handling

The interface includes comprehensive error handling:

```tsx
try {
  const response = await fetch(url, {
    method,
    headers: { "Content-Type": "application/json" },
    body: body ? JSON.stringify(body) : undefined,
  });

  const data = (await response.json()) as unknown;

  if (!response.ok) {
    throw new Error(
      typeof data === "object" &&
        data !== null &&
        "error" in data &&
        typeof (data as { error?: unknown }).error === "string"
        ? (data as { error: string }).error
        : "Request failed",
    );
  }

  setResult({ title, payload: data });
} catch (err) {
  setError(err instanceof Error ? err.message : "Unexpected error");
} finally {
  setLoading(null);
}
```

**Error scenarios handled:**

- Network failures
- API errors with descriptive messages
- Invalid JSON in request payload
- Missing required fields (payment ID, proof token, etc.)
