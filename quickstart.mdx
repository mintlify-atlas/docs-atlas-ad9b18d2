Get your GalioPay integration up and running in just a few minutes. This guide will walk you through installation, configuration, and creating your first payment link.

<Steps>
  <Step title="Install Dependencies">
    Clone or download the integration example and install the required packages:

    ```bash
    npm install
    ```

    The project uses:
    - **Next.js 16.1.6** (App Router)
    - **React 19**
    - **TypeScript**
    - **Tailwind CSS 4**
  </Step>

  <Step title="Configure Environment Variables">
    Create your environment file from the example:

    ```bash
    cp .env.example .env.local
    ```

    Add your GalioPay credentials to `.env.local`:

    ```env
    GALIOPAY_API_BASE=https://pay.galio.app/api
    GALIOPAY_API_KEY=your_galiopay_api_key
    GALIOPAY_CLIENT_ID=your_galiopay_client_id
    APP_URL=http://localhost:3000
    ```

    <Note>
      **Where to find your credentials:**
      - `GALIOPAY_API_KEY`: Your secret API key from the GalioPay dashboard
      - `GALIOPAY_CLIENT_ID`: Your client identifier from the GalioPay dashboard
      - `APP_URL`: Your application URL (use `http://localhost:3000` for local development)
    </Note>

    <Warning>
      Never expose your `GALIOPAY_API_KEY` in client-side code. All GalioPay API calls should be made from server-side routes only.
    </Warning>
  </Step>

  <Step title="Start the Development Server">
    Run the development server:

    ```bash
    npm run dev
    ```

    Your application will be available at:
    - Main app: `http://localhost:3000`
    - Interactive examples: `http://localhost:3000/examples`
  </Step>

  <Step title="Create Your First Payment Link">
    Now let's create a payment link. This example shows the server-side implementation that keeps your credentials secure.

    ### Set up the API route

    Create a server-side API route at `src/app/api/checkout/payment-link/route.ts`:

    ```typescript src/app/api/checkout/payment-link/route.ts
    import { NextResponse } from "next/server";
    import { createPaymentLink } from "@/lib/galiopay/client";
    import type { CreatePaymentLinkInput } from "@/lib/galiopay/types";

    export async function POST(request: Request) {
      try {
        const body = await request.json();

        const appUrl = process.env.APP_URL ?? "http://localhost:3000";

        const payload: CreatePaymentLinkInput = {
          items: [
            {
              title: `Order ${body.orderId}`,
              quantity: 1,
              unitPrice: body.amount,
              currencyId: "ARS",
            },
          ],
          referenceId: body.orderId,
          description: "Payment from Next.js integration example",
          backUrl: {
            success: `${appUrl}/checkout/success?orderId=${body.orderId}`,
            failure: `${appUrl}/checkout/failure?orderId=${body.orderId}`,
          },
        };

        const result = await createPaymentLink(payload);

        return NextResponse.json(result, { status: 201 });
      } catch (error) {
        console.error("Error creating payment link:", error);

        const message =
          error instanceof Error ? error.message : "Unexpected server error";
        return NextResponse.json({ error: message }, { status: 500 });
      }
    }
    ```

    ### Use the GalioPay client

    The integration includes a pre-built client at `src/lib/galiopay/client.ts` that handles authentication and requests:

    ```typescript src/lib/galiopay/client.ts
    import type { CreatePaymentLinkInput, CreatePaymentLinkResponse } from "@/lib/galiopay/types";

    function getConfig() {
      const apiBase = process.env.GALIOPAY_API_BASE;
      const apiKey = process.env.GALIOPAY_API_KEY;
      const clientId = process.env.GALIOPAY_CLIENT_ID;

      if (!apiBase || !apiKey || !clientId) {
        throw new Error("Missing required GalioPay environment variables");
      }

      return { apiBase, apiKey, clientId };
    }

    async function requestGalioPay<T>(
      path: string,
      init: RequestInit,
      withAuth = true,
    ): Promise<T> {
      const { apiBase, apiKey, clientId } = getConfig();
      const headers: HeadersInit = {
        "Content-Type": "application/json",
        ...(withAuth
          ? {
              Authorization: `Bearer ${apiKey}`,
              "x-client-id": clientId,
            }
          : {}),
        ...(init.headers ?? {}),
      };

      const response = await fetch(`${apiBase}${path}`, {
        ...init,
        headers,
        cache: "no-store",
      });

      const data = await response.json();

      if (!response.ok) {
        throw new Error(data.error || "GalioPay request failed");
      }

      return data as T;
    }

    export async function createPaymentLink(
      payload: CreatePaymentLinkInput,
    ): Promise<CreatePaymentLinkResponse> {
      return requestGalioPay<CreatePaymentLinkResponse>("/payment-links", {
        method: "POST",
        body: JSON.stringify(payload),
      });
    }
    ```

    <Tip>
      The client automatically adds authentication headers (`Authorization` and `x-client-id`) to all requests that require credentials.
    </Tip>

    ### Test the integration

    Call your API route from a client component:

    ```typescript
    const response = await fetch('/api/checkout/payment-link', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        orderId: 'order-123',
        amount: 5000 // Amount in cents (ARS 50.00)
      })
    });

    const data = await response.json();
    // data.url contains the payment link
    window.location.href = data.url;
    ```

    The response will contain a `url` field with the payment link where users can complete their payment.
  </Step>
</Steps>

## Architecture Overview

The integration follows a secure server-side pattern:

1. **Client-side UI** (`src/app/checkout/page.tsx`) - Collects payment information
2. **API Route** (`src/app/api/checkout/payment-link/route.ts`) - Handles server-side logic
3. **GalioPay Client** (`src/lib/galiopay/client.ts`) - Manages authenticated API requests
4. **GalioPay API** - Processes payments securely

<Note>
  This architecture ensures your API credentials never reach the browser, maintaining security while providing a smooth user experience.
</Note>
